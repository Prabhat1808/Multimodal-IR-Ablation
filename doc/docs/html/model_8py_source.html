18 def map_rank(traingnd, testgnd, hamming_rank):
19  """
20  This funtion returns map@all metric score.
21  hamming_rank : numtrain x numtest
22  *gnd : numsamples x labelsize
23  """
24  numtrain, numtest = hamming_rank.shape
25  apall = np.zeros((numtrain, numtest))
26  patk = np.zeros((numtrain, numtest))
27  ratk = np.zeros((numtrain, numtest))
28  aa = np.array([i+1 for i in range(numtrain)])
29  for i in range(numtest):
30  y = hamming_rank[:, i]
31  new_label = np.array([0 for j in range(numtrain)])
32  relevant_indices = (np.matmul(traingnd, testgnd[i, :].reshape((-1, 1))) > 0).reshape(-1)
33  new_label[relevant_indices] = 1
34  total_relevant = np.sum(new_label)
35  xx = np.cumsum(new_label[y]) #retrieved relevant
36  ratk[:, i] = xx / total_relevant
37  patk[:, i] = xx / aa
38  x = xx * new_label[y]
39  p = x / aa #precision@k
40  p = np.cumsum(p)
41  mask = (p != 0)
42  p[mask] = p[mask]/xx[mask]
43  apall[:, i] = p.copy()
44  pre = np.mean(patk, axis=1)
45  recall = np.mean(ratk, axis=1)
46  mAP = np.mean(apall, axis=1)
47  return mAP, pre, recall
